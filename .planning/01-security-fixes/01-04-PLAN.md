```python
"""
SSRF (Server-Side Request Forgery) Protection Module.

Validates URLs before server-side fetching to prevent access to internal resources.
"""

import ipaddress
import socket
from urllib.parse import urlparse
from typing import Optional
import httpx


class SSRFError(Exception):
    """Raised when SSRF attempt is detected."""
    pass


# Private IP ranges to block (RFC 1918 + others)
PRIVATE_IP_RANGES = [
    ipaddress.ip_network('10.0.0.0/8'),      # RFC 1918 private
    ipaddress.ip_network('172.16.0.0/12'),   # RFC 1918 private
    ipaddress.ip_network('192.168.0.0/16'),  # RFC 1918 private
    ipaddress.ip_network('127.0.0.0/8'),     # Loopback
    ipaddress.ip_network('169.254.0.0/16'),  # Link-local (AWS metadata)
    ipaddress.ip_network('::1/128'),         # IPv6 loopback
    ipaddress.ip_network('fc00::/7'),        # IPv6 ULA
    ipaddress.ip_network('fe80::/10'),       # IPv6 link-local
]

ALLOWED_SCHEMES = {'http', 'https'}

BLOCKED_HOSTS = {
    'localhost',
    'localhost.localdomain',
    'ip6-localhost',
    'ip6-loopback',
    'metadata.google.internal',  # GCP metadata
    'metadata',                   # Azure metadata
}


def validate_url_for_ssrf(url: str) -> str:
    """
    Validate URL is safe for server-side fetching.
    
    Returns:
        The validated URL
        
    Raises:
        SSRFError: If URL is potentially malicious
    """
    try:
        parsed = urlparse(url)
    except Exception as e:
        raise SSRFError(f"Invalid URL format: {e}")
    
    # Check scheme
    if parsed.scheme.lower() not in ALLOWED_SCHEMES:
        raise SSRFError(f"URL scheme not allowed: {parsed.scheme}. Only HTTP and HTTPS are allowed.")
    
    # Check for blocked hostnames
    hostname = parsed.hostname
    if not hostname:
        raise SSRFError("URL must have a hostname")
    
    if hostname.lower() in BLOCKED_HOSTS:
        raise SSRFError(f"Hostname not allowed: {hostname}")
    
    # Resolve hostname to IP and check for private ranges
    try:
        # Get all IPs for hostname (handles DNS rebinding)
        port = parsed.port or (443 if parsed.scheme == 'https' else 80)
        addr_infos = socket.getaddrinfo(hostname, port)
        
        for family, socktype, proto, canonname, sockaddr in addr_infos:
            ip_str = sockaddr[0]
            try:
                ip = ipaddress.ip_address(ip_str)
            except ValueError:
                continue  # Skip invalid IPs
            
            for private_range in PRIVATE_IP_RANGES:
                if ip in private_range:
                    raise SSRFError(
                        f"Resolved IP {ip} is in private range. "
                        f"Access to internal resources is blocked."
                    )
                    
    except socket.gaierror as e:
        raise SSRFError(f"Could not resolve hostname: {e}")
    
    return url


def fetch_webpage_safe(url: str, timeout: float = 10.0) -> str:
    """
    Safely fetch webpage content with SSRF protection.
    
    Args:
        url: URL to fetch
        timeout: Request timeout in seconds
        
    Returns:
        Page content as text (truncated to 5000 chars)
    """
    from bs4 import BeautifulSoup
    
    # Validate URL first
    try:
        validated_url = validate_url_for_ssrf(url)
    except SSRFError as e:
        return f"SSRF protection blocked this URL: {str(e)}"
    
    try:
        # Do NOT follow redirects - they could bypass validation
        resp = httpx.get(validated_url, timeout=timeout, follow_redirects=False)
        resp.raise_for_status()
        
        soup = BeautifulSoup(resp.text, "html.parser")
        content = soup.get_text(separator="\n", strip=True)
        
        # Limit size
        if len(content) > 5000:
            return content[:5000] + "\n... [TRUNCATED]"
        return content
        
    except httpx.HTTPStatusError as e:
        return f"HTTP error fetching webpage: {e}"
    except httpx.RequestError as e:
        return f"Failed to fetch webpage: {str(e)}"
    except Exception as e:
        return f"Error fetching webpage: {str(e)}"
```

import pytest
from security.ssrf import SSRFError, validate_url_for_ssrf, fetch_webpage_safe


class TestSSRFProtection:
    """Tests for SSRF protection."""
    
    def test_private_ip_blocked(self):
        """Test that private IP ranges are blocked."""
        private_urls = [
            "http://10.0.0.1/admin",
            "http://172.16.0.1/admin",
            "http://192.168.1.1/admin",
        ]
        for url in private_urls:
            with pytest.raises(SSRFError, match="private range"):
                validate_url_for_ssrf(url)
    
    def test_localhost_blocked(self):
        """Test that localhost is blocked."""
        localhost_urls = [
            "http://localhost/admin",
            "http://127.0.0.1/admin",
            "http://[::1]/admin",  # IPv6 loopback
        ]
        for url in localhost_urls:
            with pytest.raises(SSRFError):
                validate_url_for_ssrf(url)
    
    def test_link_local_blocked(self):
        """Test that link-local addresses are blocked (AWS metadata)."""
        with pytest.raises(SSRFError):
            validate_url_for_ssrf("http://169.254.169.254/latest/meta-data/")
    
    def test_cloud_metadata_blocked(self):
        """Test that cloud metadata hostnames are blocked."""
        metadata_urls = [
            "http://metadata.google.internal/",
            "http://metadata/",
        ]
        for url in metadata_urls:
            with pytest.raises(SSRFError):
                validate_url_for_ssrf(url)
    
    def test_invalid_scheme_blocked(self):
        """Test that non-HTTP schemes are blocked."""
        invalid_schemes = [
            "file:///etc/passwd",
            "ftp://example.com/file",
            "javascript:alert(1)",
        ]
        for url in invalid_schemes:
            with pytest.raises(SSRFError, match="scheme"):
                validate_url_for_ssrf(url)
    
    def test_valid_url_allowed(self):
        """Test that valid public URLs are allowed."""
        # This would make a real DNS request, so we mock or skip in CI
        # For now, just test that the URL parsing works
        valid_url = "https://example.com/page"
        # We can't test the DNS resolution without network access
        # But we can test the parsing
        from urllib.parse import urlparse
        parsed = urlparse(valid_url)
        assert parsed.scheme == "https"
        assert parsed.hostname == "example.com"


class TestFetchWebpage:
    """Tests for fetch_webpage_safe function."""
    
    def test_ssrf_blocked_returns_error_message(self):
        """Test that blocked URLs return error message, not exception."""
        result = fetch_webpage_safe("http://localhost/admin")
        assert "SSRF protection blocked" in result
    
    def test_private_ip_blocked(self):
        """Test that private IP fetch returns error."""
        result = fetch_webpage_safe("http://192.168.1.1/admin")
        assert "SSRF protection blocked" in result
