## Current Vulnerability

**File:** `src/security/taint_engine.py:32-41`

**Issue:** Uses SHA256 hash without secret key - tokens can be forged.

```python
# VULNERABLE CODE:
def generate_approval_token(...):
    raw = f"{owner_user_id}:{task_id}:..."
    return hashlib.sha256(raw.encode()).hexdigest()  # No secret!
```

**Attack vector:** Attacker can compute valid tokens by knowing the public parameters.

## Research Solution

From RESEARCH.md Pattern 2:
- Use `hmac` module with secret key
- Use `secrets.token_urlsafe()` for nonces
- Use `hmac.compare_digest()` for timing-attack resistance

2. Create `ApprovalTokenManager` class (from RESEARCH.md Pattern 2):
   ```python
   class ApprovalTokenManager:
       """HMAC-based approval token generation and verification."""
       
       def __init__(self, secret_key: Optional[bytes] = None):
           from config.settings import SecurityConfig
           config = SecurityConfig()
           self._secret_key = secret_key or config.approval_secret_key.encode()
       
       def generate_token(
           self,
           owner_user_id: str,
           task_id: str,
           approval_id: str,
           action_hash: str,
           expires_in: int = 3600,
           nonce: Optional[str] = None,
           channel_binding_hash: str = ""
       ) -> str:
           """Generate HMAC-signed approval token."""
           exp = int(time.time()) + expires_in
           nonce = nonce or secrets.token_urlsafe(16)
           
           # Build message to sign
           message = f"{owner_user_id}:{task_id}:{approval_id}:{action_hash}:{exp}:{nonce}:{channel_binding_hash}"
           
           # Generate HMAC signature
           signature = hmac.new(
               self._secret_key,
               message.encode('utf-8'),
               hashlib.sha256
           ).hexdigest()
           
           return f"{exp}:{nonce}:{signature}"
       
       def verify_token(
           self,
           token: str,
           owner_user_id: str,
           task_id: str,
           approval_id: str,
           action_hash: str,
           channel_binding_hash: str = ""
       ) -> bool:
           """Verify token with constant-time comparison."""
           try:
               parts = token.split(':')
               if len(parts) != 3:
                   return False
               
               exp_str, nonce, provided_signature = parts
               exp = int(exp_str)
               
               # Check expiration
               if time.time() > exp:
                   return False
               
               # Recompute signature
               message = f"{owner_user_id}:{task_id}:{approval_id}:{action_hash}:{exp}:{nonce}:{channel_binding_hash}"
               expected_signature = hmac.new(
                   self._secret_key,
                   message.encode('utf-8'),
                   hashlib.sha256
               ).hexdigest()
               
               # CRITICAL: Use constant-time comparison
               return hmac.compare_digest(expected_signature, provided_signature)
               
           except (ValueError, TypeError):
               return False
   ```

```python
import time

def test_timing_attack_resistance():
    """Verify token comparison uses constant-time algorithm."""
    from src.security.taint_engine import ApprovalTokenManager
    
    manager = ApprovalTokenManager()
    
    # Generate valid token
    valid_token = manager.generate_token("user1", "task1", "approval1", "hash1")
    
    # Completely wrong token
    wrong_token = "1234567890:abcdef:wrong_signature_here_12345"
    
    # Measure verification times
    times_valid = []
    times_wrong = []
    
    for _ in range(100):
        start = time.perf_counter()
        manager.verify_token(valid_token, "user1", "task1", "approval1", "hash1")
        times_valid.append(time.perf_counter() - start)
        
        start = time.perf_counter()
        manager.verify_token(wrong_token, "user1", "task1", "approval1", "hash1")
        times_wrong.append(time.perf_counter() - start)
    
    avg_valid = sum(times_valid) / len(times_valid)
    avg_wrong = sum(times_wrong) / len(times_wrong)
    
    # Timing should be within 2x of each other (constant-time comparison)
    ratio = max(avg_valid, avg_wrong) / min(avg_valid, avg_wrong)
    assert ratio < 2.0, f"Timing variance too high: {ratio}x"
```

This test verifies that `hmac.compare_digest()` is being used, not `==`.
