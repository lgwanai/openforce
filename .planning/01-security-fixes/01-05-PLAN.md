   ```python
   @dataclass
   class TaintedValue:
       """
       A value with taint metadata.
       Tracks sources and trust level for security decisions.
       """
       value: Any
       sources: Set[TaintSource] = field(default_factory=lambda: {TaintSource.INTERNAL})
       trust_level: TrustLevel = TrustLevel.TRUSTED
       
       def __post_init__(self):
           """Derive trust level from sources."""
           if TaintSource.WEB in self.sources or TaintSource.UPLOAD in self.sources:
               self.trust_level = TrustLevel.UNTRUSTED
           elif TaintSource.USER_FREE_TEXT in self.sources or TaintSource.SEARCH in self.sources:
               self.trust_level = TrustLevel.DERIVED
       
       def propagate_to(self, new_value: Any) -> 'TaintedValue':
           """Propagate taint to a derived value."""
           return TaintedValue(
               value=new_value,
               sources=self.sources.copy(),
               trust_level=self.trust_level
           )
       
       @staticmethod
       def trusted(value: Any) -> 'TaintedValue':
           """Create a trusted internal value."""
           return TaintedValue(value=value, sources={TaintSource.INTERNAL})
       
       @staticmethod
       def from_web(value: Any) -> 'TaintedValue':
           """Create a value from web source."""
           return TaintedValue(value=value, sources={TaintSource.WEB})
       
       @staticmethod
       def from_user(value: Any) -> 'TaintedValue':
           """Create a value from user input."""
           return TaintedValue(value=value, sources={TaintSource.USER_FREE_TEXT})
   ```

```python
class TaintEngine:
    """
    Taint propagation and sensitive tool verification.
    """
    
    # High-risk tools that require elevated trust
    HIGH_RISK_TOOLS = frozenset([
        "execute_command",
        "delete_file",
        "write_api",
        "run_shell",
    ])
    
    # Medium-risk tools that accept derived trust
    MEDIUM_RISK_TOOLS = frozenset([
        "write_file",
        "run_script",
    ])
    
    @classmethod
    def check_tool_call(
        cls,
        tool_name: str,
        args: Dict[str, Any],
        tainted_args: Optional[Dict[str, TaintedValue]] = None
    ) -> bool:
        """
        Check if tool call is allowed based on taint level.
        
        Returns:
            True if allowed, False if blocked (requires approval)
        """
        # High-risk tools always require approval (return False)
        if tool_name in cls.HIGH_RISK_TOOLS:
            return False
        
        # Medium-risk tools check taint level
        if tool_name in cls.MEDIUM_RISK_TOOLS:
            if tainted_args:
                for arg_name, tainted_value in tainted_args.items():
                    if tainted_value.trust_level == TrustLevel.UNTRUSTED:
                        # Untrusted data in medium-risk tool requires approval
                        return False
        
        # Low-risk tools or trusted data: allowed
        return True
    
    @staticmethod
    def get_trust_level(sources: List[str]) -> str:
        """Determine trust level from data sources (backwards compatible)."""
        source_set = set(sources)
        
        if "web" in source_set or "upload" in source_set:
            return "Untrusted"
        if "user_free_text" in source_set or "search" in source_set:
            return "Derived"
        return "Trusted"
    
    @staticmethod
    def sanitize(value: TaintedValue, sanitizer: str) -> TaintedValue:
        """
        Sanitize a tainted value and upgrade its trust level.
        
        Args:
            value: The tainted value
            sanitizer: Name of sanitization method (for audit)
        
        Returns:
            New TaintedValue with DERIVED trust level
        """
        return TaintedValue(
            value=value.value,
            sources=value.sources | {TaintSource.INTERNAL},
            trust_level=TrustLevel.DERIVED
        )
```

```python
def taint_source(source: TaintSource):
    """
    Decorator to mark function output as tainted from a specific source.
    
    Usage:
        @taint_source(TaintSource.WEB)
        def fetch_webpage(url: str) -> TaintedValue:
            ...
    """
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = func(*args, **kwargs)
            # If already a TaintedValue, return as-is
            if isinstance(result, TaintedValue):
                return result
            # Wrap in TaintedValue with specified source
            return TaintedValue(value=result, sources={source})
        return wrapper
    return decorator
```

This allows tools to be marked as producing tainted data:

```python
@taint_source(TaintSource.WEB)
def fetch_webpage(url: str) -> TaintedValue:
    ...
```

```python
        """Test creating trusted internal value."""

        assert val.trust_level == TrustLevel.TRUSTED
        assert TaintSource.INTERNAL in val.sources
    
    def test_web_value_is_untrusted(self):
        """Test that web source creates untrusted value."""
        val = TaintedValue.from_web("web content")
        assert val.trust_level == TrustLevel.UNTRUSTED
        assert TaintSource.WEB in val.sources

        """Test that user input creates derived trust value."""

        """Test that taint propagates through transformations."""

        assert TaintSource.WEB in derived.sources

        assert sanitized.trust_level == TrustLevel.DERIVED  # Upgraded
        assert TaintSource.INTERNAL in sanitized.sources  # Internal source added
```
