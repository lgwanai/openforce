    """
    Security audit tools for Duchayuan Agent.

    Handles security scanning, secret detection, and vulnerability checks.
    """

    import os
    import re
    import subprocess
    from pathlib import Path
    from typing import List, Dict, Any
    import json


    # Common secret patterns
    SECRET_PATTERNS = {
        "api_key": r"(?i)(api[_-]?key|apikey)\s*[=:]\s*['\"][^'\"]+['\"]",
        "secret_key": r"(?i)(secret[_-]?key|secretkey)\s*[=:]\s*['\"][^'\"]+['\"]",
        "password": r"(?i)(password|passwd|pwd)\s*[=:]\s*['\"][^'\"]+['\"]",
        "token": r"(?i)(token|auth[_-]?token)\s*[=:]\s*['\"][^'\"]+['\"]",
        "private_key": r"-----BEGIN (RSA |EC )?PRIVATE KEY-----",
        "aws_key": r"(?:A3T[A-Z0-9]|AKIA|AGPA|AIDA|AROA|AIPA|ANPA|ANVA|ASIA)[0-9A-Z]{16}",
        "github_token": r"(ghp|gho|ghu|ghs|ghr)_[A-Za-z0-9_]{36,255}",
    }


    def security_scan(path: str = "src/", severity: str = "all") -> str:
        """
        Run security scan using bandit.

        Args:
            path: Path to scan
            severity: Minimum severity level (all, low, medium, high)

        Returns:
            Security scan results
        """
        try:
            cmd = ["bandit", "-r", path, "-f", "json"]

            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=120
            )

            # Parse results
            if result.stdout:
                data = json.loads(result.stdout)
                findings = []

                for issue in data.get("results", []):
                    if severity == "all" or issue.get("issue_severity", "").lower() == severity.lower():
                        findings.append({
                            "severity": issue.get("issue_severity"),
                            "confidence": issue.get("issue_confidence"),
                            "test_id": issue.get("test_id"),
                            "test_name": issue.get("test_name"),
                            "file": issue.get("filename"),
                            "line": issue.get("line_number"),
                            "message": issue.get("issue_text")
                        })

                return json.dumps(findings, indent=2, ensure_ascii=False)

            return "No security issues found"

        except subprocess.TimeoutExpired:
            return "Security scan timed out"
        except FileNotFoundError:
            return "bandit not installed - pip install bandit"
        except json.JSONDecodeError:
            return result.stdout or "No output from bandit"
        except Exception as e:
            return f"Error: {str(e)}"


    def check_secrets(path: str = "src/") -> str:
        """
        Check for hardcoded secrets.

        Args:
            path: Path to scan for secrets

        Returns:
            Secret detection results
        """
        try:
            project_root = Path.cwd()
            scan_path = (project_root / path).resolve()

            if not scan_path.exists():
                return f"Path not found: {path}"

            findings = []

            # Walk through all files
            for file_path in scan_path.rglob("*"):
                if file_path.is_file() and file_path.suffix in [".py", ".js", ".ts", ".json", ".yaml", ".yml", ".env"]:
                    try:
                        content = file_path.read_text(encoding="utf-8", errors="ignore")

                        for secret_type, pattern in SECRET_PATTERNS.items():
                            matches = re.finditer(pattern, content, re.MULTILINE)
                            for match in matches:
                                # Get line number
                                line_num = content[:match.start()].count('\n') + 1
                                findings.append({
                                    "type": secret_type,
                                    "file": str(file_path.relative_to(project_root)),
                                    "line": line_num,
                                    "matched": match.group()[:50] + "..." if len(match.group()) > 50 else match.group()
                                })

                    except Exception:
                        continue

            if findings:
                return json.dumps(findings, indent=2, ensure_ascii=False)
            return "No secrets detected"

        except Exception as e:
            return f"Error: {str(e)}"


    def check_dependencies(path: str = ".") -> str:
        """
        Check dependencies for known vulnerabilities.

        Args:
            path: Path to project root

        Returns:
            Dependency vulnerability report
        """
        try:
            project_root = Path.cwd()
            req_file = project_root / path / "requirements.txt"

            if not req_file.exists():
                return "No requirements.txt found"

            # Run pip-audit if available
            result = subprocess.run(
                ["pip-audit", "-r", str(req_file), "--format=json"],
                capture_output=True,
                text=True,
                timeout=60
            )

            if result.returncode == 0 and result.stdout:
                return result.stdout
            elif "pip-audit" in result.stderr:
                return "pip-audit not installed - pip install pip-audit"
            else:
                return result.stdout or "No vulnerabilities found"

        except FileNotFoundError:
            return "pip-audit not installed - pip install pip-audit"
        except subprocess.TimeoutExpired:
            return "Dependency check timed out"
        except Exception as e:
            return f"Error: {str(e)}"


    def generate_security_report(path: str = "src/") -> str:
        """
        Generate comprehensive security report.

        Args:
            path: Path to scan

        Returns:
            Security report in markdown format
        """
        report_lines = [
            "# Security Audit Report",
            "",
            "## Security Scan Results",
            "",
            security_scan(path),
            "",
            "## Secret Detection Results",
            "",
            check_secrets(path),
            "",
            "## Dependency Check",
            "",
            check_dependencies("."),
        ]

        return "\n".join(report_lines)
