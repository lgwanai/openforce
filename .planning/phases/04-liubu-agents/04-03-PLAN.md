    ```python
    """
    Code review and testing tools for Xingbu Agent.

    Handles code review, test execution, and coverage checking.
    """

    import subprocess
    import os
    from pathlib import Path
    from typing import Optional


    def review_code(filepath: str, focus: str = "all") -> str:
        """
        Review code file for issues.

        Args:
            filepath: Path to file to review
            focus: Focus area (security, style, performance, all)

        Returns:
            Review findings
        """
        try:
            project_root = Path.cwd()
            absolute_path = (project_root / filepath).resolve()

            if not absolute_path.exists():
                return f"File not found: {filepath}"

            # Run ruff for linting
            result = subprocess.run(
                ["ruff", "check", str(absolute_path)],
                capture_output=True,
                text=True,
                timeout=30
            )

            findings = []
            if result.stdout:
                findings.append(f"Ruff findings:\n{result.stdout}")

            # Run mypy for type checking if Python file
            if filepath.endswith(".py"):
                mypy_result = subprocess.run(
                    ["mypy", str(absolute_path), "--ignore-missing-imports"],
                    capture_output=True,
                    text=True,
                    timeout=30
                )
                if mypy_result.stdout:
                    findings.append(f"Type check findings:\n{mypy_result.stdout}")

            return "\n\n".join(findings) if findings else "No issues found"

        except subprocess.TimeoutExpired:
            return "Review timed out"
        except FileNotFoundError:
            return "Linter not available (ruff/mypy not installed)"
        except Exception as e:
            return f"Error: {str(e)}"


    def run_tests(test_path: str = "tests/", pattern: str = "") -> str:
        """
        Run tests in specified path.

        Args:
            test_path: Path to test directory or file
            pattern: Test pattern to match

        Returns:
            Test results
        """
        try:
            cmd = ["pytest", test_path, "-v", "--tb=short"]
            if pattern:
                cmd.extend(["-k", pattern])

            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                timeout=120
            )

            output = result.stdout
            if result.stderr and "warning" not in result.stderr.lower():
                output += f"\n{result.stderr}"

            return output or "(no output)"

        except subprocess.TimeoutExpired:
            return "Tests timed out (120s limit)"
        except FileNotFoundError:
            return "pytest not installed"
        except Exception as e:
            return f"Error: {str(e)}"


    def check_coverage(test_path: str = "tests/", source_path: str = "src/") -> str:
        """
        Check test coverage.

        Args:
            test_path: Path to tests
            source_path: Path to source code

        Returns:
            Coverage report
        """
        try:
            result = subprocess.run(
                ["pytest", test_path, f"--cov={source_path}", "--cov-report=term-missing"],
                capture_output=True,
                text=True,
                timeout=120
            )

            return result.stdout or "(no output)"

        except subprocess.TimeoutExpired:
            return "Coverage check timed out"
        except FileNotFoundError:
            return "pytest-cov not installed"
        except Exception as e:
            return f"Error: {str(e)}"


    def run_security_scan(path: str = "src/") -> str:
        """
        Run security scan using bandit.

        Args:
            path: Path to scan

        Returns:
            Security scan results
        """
        try:
            result = subprocess.run(
                ["bandit", "-r", path, "-f", "txt"],
                capture_output=True,
                text=True,
                timeout=60
            )

            return result.stdout or result.stderr or "(no issues found)"

        except subprocess.TimeoutExpired:
            return "Security scan timed out"
        except FileNotFoundError:
            return "bandit not installed"
        except Exception as e:
            return f"Error: {str(e)}"
    ```
