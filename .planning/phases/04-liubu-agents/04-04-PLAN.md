    """
    Documentation generation tools for Libu2 Agent.

    Handles docstring extraction, markdown formatting, and doc generation.
    """

    import os
    import ast
    from pathlib import Path
    from typing import List, Dict, Any, Optional
    import json


    def extract_docstrings(filepath: str) -> str:
        """
        Extract docstrings from a Python file.

        Args:
            filepath: Path to Python file

        Returns:
            JSON with extracted docstrings
        """
        try:
            project_root = Path.cwd()
            absolute_path = (project_root / filepath).resolve()

            if not absolute_path.exists():
                return f"File not found: {filepath}"

            with open(absolute_path, "r", encoding="utf-8") as f:
                source = f.read()

            tree = ast.parse(source)

            docs = []

            for node in ast.walk(tree):
                if isinstance(node, ast.FunctionDef):
                    doc = ast.get_docstring(node)
                    if doc:
                        docs.append({
                            "type": "function",
                            "name": node.name,
                            "docstring": doc,
                            "line": node.lineno
                        })

                elif isinstance(node, ast.ClassDef):
                    doc = ast.get_docstring(node)
                    if doc:
                        docs.append({
                            "type": "class",
                            "name": node.name,
                            "docstring": doc,
                            "line": node.lineno
                        })

            return json.dumps(docs, indent=2, ensure_ascii=False)

        except SyntaxError as e:
            return f"Syntax error in file: {str(e)}"
        except Exception as e:
            return f"Error: {str(e)}"


    def generate_doc(module_name: str, output_format: str = "markdown") -> str:
        """
        Generate documentation for a module.

        Args:
            module_name: Name of the module (e.g., "src.agents.bingbu")
            output_format: Output format (markdown, html, json)

        Returns:
            Generated documentation
        """
        try:
            # Convert module name to file path
            filepath = module_name.replace(".", "/") + ".py"
            project_root = Path.cwd()
            absolute_path = (project_root / filepath).resolve()

            if not absolute_path.exists():
                return f"Module not found: {module_name}"

            docs = extract_docstrings(str(absolute_path))

            if output_format == "json":
                return docs

            # Parse and format as markdown
            doc_data = json.loads(docs)

            lines = [f"# {module_name}", "", "## Documentation", ""]

            for item in doc_data:
                if item["type"] == "class":
                    lines.append(f"### Class: `{item['name']}`")
                else:
                    lines.append(f"#### Function: `{item['name']}`")

                lines.append("")
                lines.append(item["docstring"])
                lines.append("")

            return "\n".join(lines)

        except Exception as e:
            return f"Error: {str(e)}"


    def format_markdown(content: str, style: str = "standard") -> str:
        """
        Format markdown content.

        Args:
            content: Markdown content to format
            style: Formatting style

        Returns:
            Formatted markdown
        """
        try:
            # Basic formatting: ensure proper spacing
            lines = content.split("\n")
            formatted = []

            for i, line in enumerate(lines):
                formatted.append(line)

                # Add blank line after headers
                if line.startswith("#") and i + 1 < len(lines) and lines[i + 1]:
                    if not lines[i + 1].startswith("#"):
                        formatted.append("")

            return "\n".join(formatted)

        except Exception as e:
            return f"Error: {str(e)}"


    def create_readme(title: str, description: str, sections: str = "") -> str:
        """
        Create a README file content.

        Args:
            title: README title
            description: Project description
            sections: JSON string of additional sections

        Returns:
            README content
        """
        try:
            lines = [
                f"# {title}",
                "",
                description,
                ""
            ]

            if sections:
                try:
                    sections_data = json.loads(sections)
                    for section in sections_data:
                        lines.append(f"## {section.get('title', 'Section')}")
                        lines.append("")
                        lines.append(section.get('content', ''))
                        lines.append("")
                except json.JSONDecodeError:
                    pass

            return "\n".join(lines)

        except Exception as e:
            return f"Error: {str(e)}"
