    """
    Skill management tools for Libu Agent.

    Handles installation, updates, and listing of skill packages.
    """

    import subprocess
    import json
    from typing import List, Dict, Any


    def install_skill(skill_name: str, source: str = "pip") -> str:
        """
        Install a skill package.

        Args:
            skill_name: Name of the skill package
            source: Installation source (pip, git, local)

        Returns:
            Installation result message
        """
        try:
            if source == "pip":
                result = subprocess.run(
                    ["pip", "install", skill_name],
                    capture_output=True,
                    text=True,
                    timeout=120
                )
            elif source == "git":
                result = subprocess.run(
                    ["pip", "install", f"git+{skill_name}"],
                    capture_output=True,
                    text=True,
                    timeout=180
                )
            else:
                return f"Unknown source: {source}"

            if result.returncode == 0:
                return f"Successfully installed: {skill_name}"
            else:
                return f"Installation failed: {result.stderr}"

        except subprocess.TimeoutExpired:
            return "Installation timed out (120s limit)"
        except Exception as e:
            return f"Error: {str(e)}"


    def update_skill(skill_name: str) -> str:
        """
        Update an installed skill package.

        Args:
            skill_name: Name of the skill to update

        Returns:
            Update result message
        """
        try:
            result = subprocess.run(
                ["pip", "install", "--upgrade", skill_name],
                capture_output=True,
                text=True,
                timeout=120
            )

            if result.returncode == 0:
                return f"Successfully updated: {skill_name}"
            else:
                return f"Update failed: {result.stderr}"

        except subprocess.TimeoutExpired:
            return "Update timed out (120s limit)"
        except Exception as e:
            return f"Error: {str(e)}"


    def list_skills() -> str:
        """
        List all installed skill packages.

        Returns:
            JSON list of installed packages
        """
        try:
            result = subprocess.run(
                ["pip", "list", "--format=json"],
                capture_output=True,
                text=True,
                timeout=30
            )

            if result.returncode == 0:
                packages = json.loads(result.stdout)
                # Filter for skill-related packages (customize as needed)
                skills = [
                    {"name": p["name"], "version": p["version"]}
                    for p in packages
                ]
                return json.dumps(skills, indent=2)
            else:
                return f"Failed to list packages: {result.stderr}"

        except Exception as e:
            return f"Error: {str(e)}"


    def uninstall_skill(skill_name: str) -> str:
        """
        Uninstall a skill package.

        Args:
            skill_name: Name of the skill to uninstall

        Returns:
            Uninstallation result message
        """
        try:
            result = subprocess.run(
                ["pip", "uninstall", "-y", skill_name],
                capture_output=True,
                text=True,
                timeout=60
            )

            if result.returncode == 0:
                return f"Successfully uninstalled: {skill_name}"
            else:
                return f"Uninstall failed: {result.stderr}"

        except Exception as e:
            return f"Error: {str(e)}"
    ```

    """
    Libu (吏部) Agent for skill management.

    This agent specializes in:
    - Installing skill packages
    - Updating skills
    - Listing available skills
    """

    from typing import TypedDict, Annotated, List, Dict, Any, Optional
    from langgraph.graph import StateGraph, END
    from langgraph.graph.message import add_messages
    from langchain_core.messages import BaseMessage, HumanMessage, AIMessage, SystemMessage, ToolMessage
    from langchain_core.tools import tool
    import json
    import os

    from src.core.config import get_llm
    from src.core.utils import invoke_llm_with_tools
    from src.tools.base import get_current_time, get_system_info


    class LibuState(TypedDict):
        """State for Libu (Ministry of Personnel) skill management agent."""
        messages: Annotated[List[BaseMessage], add_messages]
        task_id: str
        goal: str
        skills_installed: List[str]
        skills_updated: List[str]
        errors: List[str]


    @tool
    def tool_install_skill(skill_name: str, source: str = "pip") -> str:
        """
        Install a skill package.

        Args:
            skill_name: Name of the skill package
            source: Installation source (pip, git, local)

        Returns:
            Installation result message
        """
        from src.tools.skill_manager import install_skill
        return install_skill(skill_name, source)


    @tool
    def tool_update_skill(skill_name: str) -> str:
        """
        Update an installed skill package.

        Args:
            skill_name: Name of the skill to update

        Returns:
            Update result message
        """
        from src.tools.skill_manager import update_skill
        return update_skill(skill_name)


    @tool
    def tool_list_skills() -> str:
        """
        List all installed skill packages.

        Returns:
            JSON list of installed packages
        """
        from src.tools.skill_manager import list_skills
        return list_skills()


    @tool
    def tool_uninstall_skill(skill_name: str) -> str:
        """
        Uninstall a skill package.

        Args:
            skill_name: Name of the skill to uninstall

        Returns:
            Uninstallation result message
        """
        from src.tools.skill_manager import uninstall_skill
        return uninstall_skill(skill_name)


    # Tool set for Libu
    libu_tools = [
        tool_install_skill,
        tool_update_skill,
        tool_list_skills,
        tool_uninstall_skill
    ]


    def react_node(state: LibuState):
        """React node for Libu Agent."""
        llm = get_llm("libu_skill")

        sys_prompt = f"""
    You are Libu (吏部), a skill management specialist.

    Current time: {get_current_time()}
    System info: {get_system_info()}

    Goal: {state["goal"]}

    Skills installed so far: {state.get("skills_installed", [])}
    Skills updated so far: {state.get("skills_updated", [])}

    Instructions:
    1. Use tool_install_skill to install new skill packages
    2. Use tool_update_skill to update existing skills
    3. Use tool_list_skills to see what's installed
    4. Use tool_uninstall_skill to remove skills if needed

    Always verify installation success after installing.
    """

        messages = [SystemMessage(content=sys_prompt)] + state["messages"]

        response = invoke_llm_with_tools(llm, libu_tools, messages)

        return {"messages": [response]}


    def tool_node(state: LibuState):
        """Execute tools for Libu Agent."""
        last_message = state["messages"][-1]
        results = []

        skills_installed = state.get("skills_installed", []).copy()
        skills_updated = state.get("skills_updated", []).copy()
        errors = state.get("errors", []).copy()

        for tool_call in last_message.tool_calls:
            tool_name = tool_call["name"]
            args = tool_call["args"]

            try:
                if tool_name == "tool_install_skill":
                    res = tool_install_skill.invoke(args)
                    skill_name = args.get("skill_name", "")
                    if "Successfully installed" in res:
                        if skill_name not in skills_installed:
                            skills_installed.append(skill_name)

                elif tool_name == "tool_update_skill":
                    res = tool_update_skill.invoke(args)
                    skill_name = args.get("skill_name", "")
                    if "Successfully updated" in res:
                        if skill_name not in skills_updated:
                            skills_updated.append(skill_name)

                elif tool_name == "tool_list_skills":
                    res = tool_list_skills.invoke(args)

                elif tool_name == "tool_uninstall_skill":
                    res = tool_uninstall_skill.invoke(args)

                else:
                    res = f"Unknown tool: {tool_name}"

            except Exception as e:
                res = f"Error: {str(e)}"
                errors.append(f"{tool_name}: {str(e)}")

            results.append(ToolMessage(
                content=str(res),
                name=tool_name,
                tool_call_id=tool_call["id"]
            ))

        return {
            "messages": results,
            "skills_installed": skills_installed,
            "skills_updated": skills_updated,
            "errors": errors
        }


    def build_libu_graph():
        """
        Build the Libu Agent graph.

        Pattern: react -> tools -> react (loop until done)
        """
        workflow = StateGraph(LibuState)

        workflow.add_node("react", react_node)
        workflow.add_node("tools", tool_node)

        workflow.set_entry_point("react")

        def should_continue(state: LibuState):
            """Check if agent should continue."""
            last_message = state["messages"][-1]

            if getattr(last_message, "tool_calls", None):
                return "tools"

            return END

        workflow.add_conditional_edges("react", should_continue)
        workflow.add_edge("tools", "react")

        return workflow.compile()
    ```
