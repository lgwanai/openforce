    """
    Environment management tools for Gongbu Agent.

    Handles virtual environment creation, command execution, and status checking.
    """

    import subprocess
    import os
    import sys
    from pathlib import Path
    from typing import Optional


    def get_envs_dir() -> Path:
        """Get the directory where environments are stored."""
        envs_dir = Path.home() / ".openforce" / "envs"
        envs_dir.mkdir(parents=True, exist_ok=True)
        return envs_dir


    def create_env(env_name: str, python_version: str = "3.11") -> str:
        """
        Create a new virtual environment.

        Args:
            env_name: Name for the environment
            python_version: Python version to use

        Returns:
            Creation result message
        """
        try:
            envs_dir = get_envs_dir()
            env_path = envs_dir / env_name

            if env_path.exists():
                return f"Environment already exists: {env_name}"

            # Create virtual environment
            result = subprocess.run(
                [sys.executable, "-m", "venv", str(env_path)],
                capture_output=True,
                text=True,
                timeout=120
            )

            if result.returncode == 0:
                return f"Created environment: {env_name} at {env_path}"
            else:
                return f"Failed to create environment: {result.stderr}"

        except subprocess.TimeoutExpired:
            return "Environment creation timed out (120s limit)"
        except Exception as e:
            return f"Error: {str(e)}"


    def run_command(command: str, env_name: str = "") -> str:
        """
        Run a command in an environment.

        Args:
            command: Command to run
            env_name: Environment name (empty for system Python)

        Returns:
            Command output
        """
        try:
            if env_name:
                envs_dir = get_envs_dir()
                env_path = envs_dir / env_name

                if not env_path.exists():
                    return f"Environment not found: {env_name}"

                # Use environment's Python
                if sys.platform == "win32":
                    python_path = env_path / "Scripts" / "python.exe"
                else:
                    python_path = env_path / "bin" / "python"

                # Run in environment
                result = subprocess.run(
                    [str(python_path), "-c", command],
                    capture_output=True,
                    text=True,
                    timeout=60
                )
            else:
                # Run with system Python
                result = subprocess.run(
                    command,
                    shell=True,
                    capture_output=True,
                    text=True,
                    timeout=60
                )

            output = result.stdout
            if result.stderr:
                output += f"\nSTDERR: {result.stderr}"

            return output or "(no output)"

        except subprocess.TimeoutExpired:
            return "Command timed out (60s limit)"
        except Exception as e:
            return f"Error: {str(e)}"


    def check_env(env_name: str) -> str:
        """
        Check environment status.

        Args:
            env_name: Name of environment to check

        Returns:
            Environment status information
        """
        try:
            envs_dir = get_envs_dir()
            env_path = envs_dir / env_name

            if not env_path.exists():
                return f"Environment not found: {env_name}"

            # Check Python version
            if sys.platform == "win32":
                python_path = env_path / "Scripts" / "python.exe"
            else:
                python_path = env_path / "bin" / "python"

            result = subprocess.run(
                [str(python_path), "--version"],
                capture_output=True,
                text=True,
                timeout=10
            )

            info = {
                "name": env_name,
                "path": str(env_path),
                "python_version": result.stdout.strip() or result.stderr.strip(),
                "exists": True
            }

            return str(info)

        except Exception as e:
            return f"Error checking environment: {str(e)}"


    def list_envs() -> str:
        """
        List all created environments.

        Returns:
            JSON list of environments
        """
        try:
            envs_dir = get_envs_dir()
            envs = []

            for env_path in envs_dir.iterdir():
                if env_path.is_dir():
                    envs.append({
                        "name": env_path.name,
                        "path": str(env_path)
                    })

            import json
            return json.dumps(envs, indent=2)

        except Exception as e:
            return f"Error listing environments: {str(e)}"


    def remove_env(env_name: str) -> str:
        """
        Remove an environment.

        Args:
            env_name: Name of environment to remove

        Returns:
            Removal result message
        """
        try:
            import shutil
            envs_dir = get_envs_dir()
            env_path = envs_dir / env_name

            if not env_path.exists():
                return f"Environment not found: {env_name}"

            shutil.rmtree(env_path)
            return f"Removed environment: {env_name}"

        except Exception as e:
            return f"Error removing environment: {str(e)}"
