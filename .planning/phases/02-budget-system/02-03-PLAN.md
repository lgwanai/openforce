from typing import Dict
from ..core.config import load_config

# Default pricing table (fallback if config missing)
DEFAULT_PRICING: Dict[str, float] = {
    "minimax-m2.5": 0.001,
    "glm-5": 0.0005,
    "gpt-4": 0.03,
    "gpt-3.5-turbo": 0.002,
    "claude-3-opus": 0.015,
    "claude-3-sonnet": 0.003,
}

_pricing_table: Dict[str, float] = {}

def get_pricing_table() -> Dict[str, float]:
    """Load pricing table from config."""
    global _pricing_table
    if _pricing_table:
        return _pricing_table

    try:
        config = load_config()
        for model_key, model_cfg in config.llm_models.items():
            model_name = model_cfg.model
            cost = getattr(model_cfg, 'cost_per_1k_tokens', None)
            if cost:
                _pricing_table[model_name] = cost
    except Exception:
        pass

    # Merge with defaults
    _pricing_table = {**DEFAULT_PRICING, **_pricing_table}
    return _pricing_table

def calculate_cost(tokens: int, model_name: str) -> float:
    """
    Calculate cost in USD for given tokens and model.

    Args:
        tokens: Number of tokens consumed
        model_name: Model identifier (e.g., "minimax-m2.5")

    Returns:
        Cost in USD
    """
    pricing = get_pricing_table()
    rate = pricing.get(model_name, 0.001)  # Default conservative rate
    return (tokens / 1000) * rate
```

async def consume_cost(self, cost_usd: float) -> None:
    """Record cost consumption."""
    async with self._lock:
        self.usage.cost_usd += cost_usd
        if self.parent:
            await self.parent.consume_cost(cost_usd)
        if exceeded := self.usage.check_exceeded(self.limits):
            self._exhausted = True
            raise BudgetExhaustedError(exceeded, self.usage)

def on_llm_end(self, response: LLMResult, **kwargs) -> None:
    """Extract token usage and calculate cost."""
    total_tokens = 0

    # Extract tokens (existing logic)
    if response.llm_output:
        token_usage = response.llm_output.get('token_usage', {})
        total_tokens = token_usage.get('total_tokens', 0)

    if not total_tokens:
        for generation_list in response.generations:
            for generation in generation_list:
                if hasattr(generation, 'generation_info'):
                    usage = generation.generation_info.get('usage', {})
                    total_tokens = usage.get('total_tokens', 0)
                    if total_tokens:
                        break

    if total_tokens:
        # Track tokens
        asyncio.create_task(self.budget_manager.consume_tokens(total_tokens))

        # Track cost if model name available
        model_name = getattr(response, 'model_name', None)
        if model_name:
            cost = calculate_cost(total_tokens, model_name)
            asyncio.create_task(self.budget_manager.consume_cost(cost))
```
