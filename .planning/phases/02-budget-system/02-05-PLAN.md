```python
from enum import Enum
from typing import List, Optional
from dataclasses import replace
from .manager import BudgetLimits


class BudgetAllocationStrategy(Enum):
    """Strategy for allocating budgets to child agents."""
    EQUAL = "equal"          # Divide equally among children
    RESERVE = "reserve"      # Keep 20% for parent, divide rest
    CUSTOM = "custom"        # User-provided ratios


def allocate_child_budgets(
    parent_limits: BudgetLimits,
    child_count: int,
    strategy: BudgetAllocationStrategy = BudgetAllocationStrategy.EQUAL,
    custom_ratios: Optional[List[float]] = None
) -> List[BudgetLimits]:
    """
    Allocate budgets for child agents.

    Time limits are NOT divided - all children share the parent's time budget.
    Token and cost limits are divided according to strategy.

    Args:
        parent_limits: Parent budget limits
        child_count: Number of child agents
        strategy: Allocation strategy
        custom_ratios: Custom ratios for CUSTOM strategy (must sum to 1.0)

    Returns:
        List of BudgetLimits, one per child agent
    """
    if child_count <= 0:
        return []

    if strategy == BudgetAllocationStrategy.EQUAL:
        return _allocate_equal(parent_limits, child_count)
    elif strategy == BudgetAllocationStrategy.RESERVE:
        return _allocate_reserve(parent_limits, child_count)
    elif strategy == BudgetAllocationStrategy.CUSTOM:
        return _allocate_custom(parent_limits, child_count, custom_ratios or [])

    return [BudgetLimits()] * child_count


def _allocate_equal(parent_limits: BudgetLimits, child_count: int) -> List[BudgetLimits]:
    """Divide parent budget equally among children."""
    return [
        BudgetLimits(
            max_tokens=parent_limits.max_tokens // child_count if parent_limits.max_tokens else None,
            max_time_seconds=parent_limits.max_time_seconds,  # Time not divided
            max_cost_usd=parent_limits.max_cost_usd / child_count if parent_limits.max_cost_usd else None,
        )
        for _ in range(child_count)
    ]


def _allocate_reserve(parent_limits: BudgetLimits, child_count: int) -> List[BudgetLimits]:
    """Keep 20% reserve for parent, divide rest equally."""
    reserve_ratio = 0.2
    available_tokens = int(parent_limits.max_tokens * (1 - reserve_ratio)) if parent_limits.max_tokens else None
    available_cost = parent_limits.max_cost_usd * (1 - reserve_ratio) if parent_limits.max_cost_usd else None

    return [
        BudgetLimits(
            max_tokens=available_tokens // child_count if available_tokens else None,
            max_time_seconds=parent_limits.max_time_seconds,  # Time not divided
            max_cost_usd=available_cost / child_count if available_cost else None,
        )
        for _ in range(child_count)
    ]


def _allocate_custom(
    parent_limits: BudgetLimits,
    child_count: int,
    ratios: List[float]
) -> List[BudgetLimits]:
    """Allocate based on custom ratios."""
    if not ratios or len(ratios) != child_count:
        # Fall back to equal if invalid ratios
        return _allocate_equal(parent_limits, child_count)

    return [
        BudgetLimits(
            max_tokens=int(parent_limits.max_tokens * ratio) if parent_limits.max_tokens else None,
            max_time_seconds=parent_limits.max_time_seconds,
            max_cost_usd=parent_limits.max_cost_usd * ratio if parent_limits.max_cost_usd else None,
        )
        for ratio in ratios
    ]
```

```python
# Minimum allocation thresholds
MIN_TOKENS_PER_CHILD = 1000
MIN_COST_PER_CHILD = 0.01  # USD


def validate_allocation(child_limits: List[BudgetLimits]) -> List[int]:
    """
    Validate allocations don't starve any child.

    Returns:
        List of child indices that would be starved (empty if all OK)
    """
    starved = []
    for i, limits in enumerate(child_limits):
        if limits.max_tokens is not None and limits.max_tokens < MIN_TOKENS_PER_CHILD:
            starved.append(i)
        if limits.max_cost_usd is not None and limits.max_cost_usd < MIN_COST_PER_CHILD:
            starved.append(i)
    return starved


def safe_allocate(
    parent_limits: BudgetLimits,
    child_count: int,
    strategy: BudgetAllocationStrategy = BudgetAllocationStrategy.EQUAL,
) -> List[BudgetLimits]:
    """
    Allocate with starvation prevention.

    If allocation would starve children, adjusts count or falls back.

    Returns:
        List of BudgetLimits with minimum viable allocations
    """
    allocations = allocate_child_budgets(parent_limits, child_count, strategy)
    starved = validate_allocation(allocations)

    if starved:
        # Reduce child count and reallocate
        viable_count = min(
            (parent_limits.max_tokens or float('inf')) // MIN_TOKENS_PER_CHILD,
            (parent_limits.max_cost_usd or float('inf')) // MIN_COST_PER_CHILD,
        )
        viable_count = max(1, int(viable_count))  # At least 1 child
        return allocate_child_budgets(parent_limits, viable_count, strategy)

    return allocations
```

This ensures no child gets less than minimum viable budget.

```python
def create_delegation_budget(
    parent_budget: BudgetManager,
    agent_type: str,
    strategy: BudgetAllocationStrategy = BudgetAllocationStrategy.RESERVE,
) -> BudgetManager:
    """
    Create a budget for delegating to a child agent.

    Args:
        parent_budget: Parent task's budget manager
        agent_type: Type of agent (e.g., "shangshu", "hubu")
        strategy: Allocation strategy

        BudgetManager for the delegated agent

    # For single delegation, use 50% of remaining budget

        parent_limits=parent_budget.limits,
        child_count=2,  # Parent + 1 child
        strategy=strategy,

    return parent_budget.allocate_child(agent_type, child_limits)
```
