    """
    Database safety utilities.

    Implements QTY-03: Context manager for DB connections
    Implements QTY-04: Atomic operations for active_task
    """

    import sqlite3
    import threading
    from typing import Optional, Callable, Any, ContextManager
    from contextlib import contextmanager
    from dataclasses import dataclass
    import logging

    logger = logging.getLogger(__name__)


    @dataclass
    class ConnectionConfig:
        """Database connection configuration."""
        path: str = "openforce.db"
        timeout: float = 30.0
        check_same_thread: bool = False


    class DatabaseConnection:
        """
        Thread-safe database connection with context manager.

        Usage:
            with DatabaseConnection() as conn:
                cursor = conn.cursor()
                cursor.execute("SELECT * FROM tasks")
        """

        _lock = threading.Lock()
        _instance = None

        def __new__(cls, config: ConnectionConfig = None):
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._config = config or ConnectionConfig()
                    cls._instance._local = threading.local()
            return cls._instance

        def _get_connection(self) -> sqlite3.Connection:
            if not hasattr(self._local, 'conn') or self._local.conn is None:
                self._local.conn = sqlite3.connect(
                    self._config.path,
                    timeout=self._config.timeout,
                    check_same_thread=self._config.check_same_thread
                )
            return self._local.conn

        def __enter__(self) -> sqlite3.Connection:
            return self._get_connection()

        def __exit__(self, exc_type, exc_val, exc_tb):
            if exc_type:
                self._get_connection().rollback()
            else:
                self._get_connection().commit()
            return False

        def close(self):
            if hasattr(self._local, 'conn') and self._local.conn:
                self._local.conn.close()
                self._local.conn = None


    @contextmanager
    def db_transaction():
        """Context manager for database transactions."""
        conn = sqlite3.connect("openforce.db")
        try:
            yield conn
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            conn.close()


    @contextmanager
    def atomic_operation(lock: threading.Lock = None):
        """
        Context manager for atomic operations.

        Usage:
            with atomic_operation() as lock:
                # Critical section
                pass
        """
        lock = lock or threading.Lock()
        lock.acquire()
        try:
            yield lock
        finally:
            lock.release()


    class AtomicCounter:
        """Thread-safe atomic counter."""

        def __init__(self, initial: int = 0):
            self._value = initial
            self._lock = threading.Lock()

        def increment(self, delta: int = 1) -> int:
            """Increment and return new value."""
            with self._lock:
                self._value += delta
                return self._value

        def decrement(self, delta: int = 1) -> int:
            """Decrement and return new value."""
            return self.increment(-delta)

        def get(self) -> int:
            """Get current value."""
            with self._lock:
                return self._value

        def set(self, value: int):
            """Set value."""
            with self._lock:
                self._value = value


    def atomic_db_update(
        table: str,
        key_column: str,
        key_value: Any,
        update_column: str,
        update_value: Any,
        condition: str = None
    ) -> bool:
        """
        Perform atomic database update.

        Args:
            table: Table name
            key_column: Column for identifying row
            key_value: Value for key column
            update_column: Column to update
            update_value: New value
            condition: Optional additional condition

        Returns: True if update succeeded
        """
        lock = threading.Lock()
        with atomic_operation(lock):
            with db_transaction() as conn:
                cursor = conn.cursor()

                sql = f"UPDATE {table} SET {update_column} = ? WHERE {key_column} = ?"
                params = [update_value, key_value]

                if condition:
                    sql += f" AND {condition}"

                cursor.execute(sql, params)
                return cursor.rowcount > 0


    def atomic_set_active_task(owner_user_id: str, task_id: Optional[str]) -> bool:
        """
        Atomically set active task for a user.

        Implements QTY-04: Atomic active_task operation
        """
        lock = threading.Lock()
        with atomic_operation(lock):
            with db_transaction() as conn:
                cursor = conn.cursor()

                if task_id is None:
                    cursor.execute(
                        "DELETE FROM active_tasks WHERE owner_user_id = ?",
                        (owner_user_id,)
                    )
                else:
                    cursor.execute('''
                        INSERT INTO active_tasks (owner_user_id, task_id)
                        VALUES (?, ?)
                        ON CONFLICT(owner_user_id) DO UPDATE SET task_id=excluded.task_id
                    ''', (owner_user_id, task_id))

                return True
